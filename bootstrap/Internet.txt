! ============================================================
! Internet  (IOSvL2 used as L3)  —  Day-0 bootstrap
! Simulated public-internet core. Pure L3 routed interfaces.
!   G0/0 -> ISP-A-HQ G0/2
!   G0/1 -> ISP-B-HQ G0/2
!   G0/2 -> ISP-A-BR G0/0
!   G0/3 -> ISP-B-BR G0/0
!   G1/0 -> NAT cloud (optional, for real-world reachability)
! ============================================================
enable
configure terminal

hostname Internet
no ip domain-lookup
ip domain name internet.net

username admin privilege 15 secret Admin@123
enable secret Admin@123

crypto key generate rsa modulus 2048
ip ssh version 2
line vty 0 4
 login local
 transport input ssh
 exec-timeout 30 0
exit

ip routing

interface GigabitEthernet0/0
 description to-ISP-A-HQ
 no switchport
 ip address 100.64.100.2 255.255.255.252
 no shutdown
exit
interface GigabitEthernet0/1
 description to-ISP-B-HQ
 no switchport
 ip address 100.64.101.2 255.255.255.252
 no shutdown
exit
interface GigabitEthernet0/2
 description to-ISP-A-BR
 no switchport
 ip address 100.64.200.1 255.255.255.252
 no shutdown
exit
interface GigabitEthernet0/3
 description to-ISP-B-BR
 no switchport
 ip address 100.64.201.1 255.255.255.252
 no shutdown
exit

! Optional: leave G1/0 shut unless you need real-world reachability.
! If NAT cloud is in use, set a static IP matching its subnet.
interface GigabitEthernet1/0
 description to-NAT-cloud
 no switchport
 shutdown
exit

interface Loopback0
 ip address 10.255.1.99 255.255.255.255
exit

ip route 100.64.10.0  255.255.255.252 100.64.100.1
ip route 100.64.20.0  255.255.255.252 100.64.101.1
ip route 100.64.110.0 255.255.255.252 100.64.200.2
ip route 100.64.120.0 255.255.255.252 100.64.201.2

end
write memory
